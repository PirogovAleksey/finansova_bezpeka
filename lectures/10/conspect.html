class MerkleProofOfReserves:
    """

    –ö–æ–∂–µ–Ω –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á –º–æ–∂–µ –ø–µ—Ä–µ–≤—ñ—Ä–∏—Ç–∏, —â–æ –π–æ–≥–æ –±–∞–ª–∞–Ω—Å –≤–∫–ª—é—á–µ–Ω–æ –≤ –¥–µ—Ä–µ–≤–æ,

    """

    @staticmethod
    def hash_node(data: str) -> str:
        return hashlib.sha256(data.encode()).hexdigest()

    @staticmethod
    def hash_pair(left: str, right: str) -> str:
        combined = left + right
        return hashlib.sha256(combined.encode()).hexdigest()

    def build_tree(self, user_balances: list) -> dict:
        """
        user_balances = [
            {"user_id": "hash_user_A", "balance": 1.5},   # 1.5 BTC
            {"user_id": "hash_user_B", "balance": 0.8},   # 0.8 BTC
            {"user_id": "hash_user_C", "balance": 3.2},   # 3.2 BTC
            {"user_id": "hash_user_D", "balance": 0.5},   # 0.5 BTC
        ]
        Total reserves: 6.0 BTC
        """

        leaves = []
        for ub in user_balances:
            leaf = self.hash_node(f"{ub['user_id']}:{ub['balance']}")
            leaves.append(leaf)

        # –ö—Ä–æ–∫ 2: –ü–æ–±—É–¥—É–≤–∞—Ç–∏ –¥–µ—Ä–µ–≤–æ –∑–Ω–∏–∑—É –≤–≥–æ—Ä—É
        #
        #              Merkle Root
        #             /            \
        #        Hash(AB)        Hash(CD)
        #       /       \       /       \
        #   Hash(A)  Hash(B) Hash(C)  Hash(D)
        #      |        |       |        |
        #  User_A:1.5 User_B:0.8 User_C:3.2 User_D:0.5
        #
        tree = [leaves]
        current_level = leaves

        while len(current_level) > 1:
            next_level = []
            for i in range(0, len(current_level), 2):
                left = current_level[i]
                right = current_level[i + 1] if i + 1 < len(current_level) else left
                next_level.append(self.hash_pair(left, right))
            tree.append(next_level)
            current_level = next_level

        merkle_root = current_level[0]
        return {"root": merkle_root, "tree": tree, "total_reserves": sum(u['balance'] for u in user_balances)}

    def generate_proof(self, tree, index: int) -> list:
        """–ì–µ–Ω–µ—Ä—É—î Merkle proof –¥–ª—è –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–≥–æ –∫–æ—Ä–∏—Å—Ç—É–≤–∞—á–∞"""
        proof = []
        for level in tree["tree"][:-1]:
            if index % 2 == 0:
                sibling_idx = index + 1 if index + 1 < len(level) else index
                proof.append({"position": "right", "hash": level[sibling_idx]})
            else:
                proof.append({"position": "left", "hash": level[index - 1]})
            index //= 2
        return proof

    def verify_proof(self, leaf_hash: str, proof: list, root: str) -> bool:
        """–ö–æ—Ä–∏—Å—Ç—É–≤–∞—á –≤–µ—Ä–∏—Ñ—ñ–∫—É—î, —â–æ –π–æ–≥–æ –±–∞–ª–∞–Ω—Å –≤–∫–ª—é—á–µ–Ω–æ –≤ –¥–µ—Ä–µ–≤–æ"""
        current = leaf_hash
        for step in proof:
            if step["position"] == "left":
                current = self.hash_pair(step["hash"], current)
            else:
                current = self.hash_pair(current, step["hash"])
        return current == root  # True = –±–∞–ª–∞–Ω—Å –ø—ñ–¥—Ç–≤–µ—Ä–¥–∂–µ–Ω–æ!
